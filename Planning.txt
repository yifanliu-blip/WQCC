            '2025-09-01', '2025-09-19', '2025-10-13', '2025-11-07',
            // Dec 22 to 26
            '2025-12-22', '2025-12-23', '2025-12-24', '2025-12-25', '2025-12-26',

            '2026-03-02', '2026-03-03', '2026-03-04', '2026-03-05', '2026-03-06', '2026-03-07', '2026-03-08', '2026-03-09',

        // Helper: Check if a date is a valid school day (Not weekend, not holiday)
        function isSchoolDay(date, isPM = false) {
            const day = date.getDay();
            if (day === 0 || day === 6) return false; // 0 = Sunday, 6 = Saturday
            if (isPM && day === 3) return false; // PM schedule takes Wednesdays off
            
            // Format to YYYY-MM-DD reliably based on local time
            const year = date.getFullYear();
            const month = String(date.getMonth() + 1).padStart(2, '0');
            const d = String(date.getDate()).padStart(2, '0');
            const dateString = `${year}-${month}-${d}`;
            
            if (holidays.has(dateString)) return false;
            return true;
        }

        // Validate Start Date on Change
        function validateStartDate() {
            if (!startDateInput.value) {
                dateError.classList.add('hidden');
                return;
            }
            
            const isPM = scheduleTimeSelect.value === 'PM';
            
            // Parse safely without timezone shifting issues
            const [y, m, d] = startDateInput.value.split('-');
            const selectedDate = new Date(y, m - 1, d);
            
            if (!isSchoolDay(selectedDate, isPM)) {
                dateError.classList.remove('hidden');
                startDateInput.value = ''; // clear invalid selection
            } else {
                dateError.classList.add('hidden');
            }
        }

        startDateInput.addEventListener('change', validateStartDate);
        scheduleTimeSelect.addEventListener('change', validateStartDate);

        // Populate Competency Dropdown
        competenciesDB.forEach(comp => {
            const option = document.createElement('option');
            option.value = comp.id;
            option.textContent = comp.title;
            competencySelect.appendChild(option);
        });

        // Update Pace Display
        studyPaceInput.addEventListener('input', (e) => {
            const val = e.target.value;
            let text = `${val}%`;
            if (val == 100) text += ' (Standard)';
            else if (val <= 70) text += ' (Fast Track)';
            else text += ' (Accelerated)';
            paceDisplay.textContent = text;
        });

        // Helper: Add only valid school days to a date
        function addSchoolDays(startDate, daysToAdd, isPM = false) {
            let currentDate = new Date(startDate);
            let daysAdded = 0;
            
            while (daysAdded < daysToAdd) {
                currentDate.setDate(currentDate.getDate() + 1);
                if (isSchoolDay(currentDate, isPM)) {
                    daysAdded++;
                }
            }
            return currentDate;
        }

        // Helper: Format Date clearly
        function formatDate(date) {
            const options = { weekday: 'short', year: 'numeric', month: 'short', day: 'numeric' };
            return date.toLocaleDateString('en-US', options);
        }

        // Generate Schedule Logic
        form.addEventListener('submit', (e) => {
            e.preventDefault();

            // 1. Gather Inputs
            const startDateVal = startDateInput.value;
            const compId = competencySelect.value;
            const paceVal = parseInt(studyPaceInput.value, 10);
            const isPM = scheduleTimeSelect.value === 'PM';

            if (!startDateVal || !compId) return;

            // Parse safely to avoid UTC offset shifting the date to the day prior
            const [y, m, d] = startDateVal.split('-');
            const startDate = new Date(y, m - 1, d);
            
            const compData = competenciesDB.find(c => c.id === compId);

            // 2. Algorithm Core
            // Base calculation: 3 hours per standard school day
            const standardDays = compData.hours / 3;
            const scheduleItems = [];
            scheduleItems.push({ 
                type: 'start', 
                title: 'Start of Competency', 
                desc: '<ul class="list-disc pl-5 mt-1 space-y-1"><li>Get textbook Accounting Principles and navigate Moodle page.</li><li>Please book a registration appointment with Yifan</li></ul>', 
                icon: 'flag', 
                dayOffset: 0 
            });

            // Branch: Use predefined structured blocks if they exist (e.g., 961-074)
            if (compData.blocks) {
                let runningStandardReducible = 0;
                let currentOffset = 0;

                compData.blocks.forEach(block => {
                    if (!block.mandatory) {
                        // Apply pace proportionally to the cumulative standard days
                        runningStandardReducible += block.standardDays;
                        let actualEndDay = Math.ceil(runningStandardReducible * (paceVal / 100));
                        currentOffset = actualEndDay - 1; // 0-indexed day offset

                        scheduleItems.push({
                            type: 'unit',
                            title: block.title,
                            desc: block.desc,
                            dayOffset: Math.max(0, currentOffset),
                            icon: 'book-open'
                        });
                    } else {
                        // Mandatory days strictly add exactly 1 day
                        currentOffset += block.standardDays;
                        scheduleItems.push({
                            type: block.type || 'pretest',
                            title: block.title,
                            desc: block.desc,
                            dayOffset: currentOffset,
                            icon: block.type === 'exam' ? 'file-text' : 'file-question'
                        });
                    }
                });
            } else {
                // Fallback Branch: Automatically distribute standard units
                const fixedDays = 2; 
                const learningDaysStandard = standardDays - fixedDays;
                
                // Apply pace ONLY to the learning days (hours minus pretest/exam)
                const learningDaysActual = Math.ceil(learningDaysStandard * (paceVal / 100));
                
                const numUnits = compData.units.length;
                const numAssignments = compData.assignments.length;
                
                if (learningDaysActual > 0) {
                    const daysPerUnit = learningDaysActual / numUnits;
                    compData.units.forEach((unit, index) => {
                        const targetDayOffset = Math.floor(daysPerUnit * index + (daysPerUnit / 2));
                        scheduleItems.push({ 
                            type: 'unit', 
                            title: unit, 
                            desc: 'Complete required reading, lectures, and practice quizzes.', 
                            dayOffset: Math.min(targetDayOffset, learningDaysActual - 1),
                            icon: 'book-open' 
                        });
                    });

                    const daysPerAssignment = learningDaysActual / numAssignments;
                    compData.assignments.forEach((assignment, index) => {
                        let targetDayOffset = Math.floor(daysPerAssignment * (index + 1)) - 1; 
                        if (index === numAssignments - 1) {
                            targetDayOffset = learningDaysActual - 1; 
                        }
                        scheduleItems.push({ 
                            type: 'assignment', 
                            title: assignment, 
                            desc: 'Submit deliverables to the evaluation portal.', 
                            dayOffset: Math.max(0, Math.min(targetDayOffset, learningDaysActual - 1)),
                            icon: 'clipboard-check' 
                        });
                    });
                }

                scheduleItems.push({
                    type: 'pretest',
                    title: 'Mandatory Pretest',
                    desc: 'Fixed 3-hour requirement (1 school day). Cannot be accelerated.',
                    dayOffset: learningDaysActual,
                    icon: 'file-question'
                });

                scheduleItems.push({
                    type: 'exam',
                    title: 'Final Examination',
                    desc: 'Fixed 3-hour requirement (1 school day). Cannot be accelerated.',
                    dayOffset: learningDaysActual + 1,
                    icon: 'file-text'
                });
            }

            // Sort timeline chronologically by calculated offset days
            scheduleItems.sort((a, b) => a.dayOffset - b.dayOffset);
            
            // Map calculated dates to the sorted items (using school days logic)
            let currentSchedule = [];
            
            // Dynamically locate the pretest item to align the strict Exam logic properly
            const pretestItem = scheduleItems.find(i => i.type === 'pretest');
            const pretestDayOffset = pretestItem ? pretestItem.dayOffset : 0;
            const pretestDate = addSchoolDays(startDate, pretestDayOffset, isPM);

            scheduleItems.forEach(item => {
                let itemDate;
                let dayDisplay;

                if (item.type === 'exam') {
                    // Exam must be on Tuesday (2) or Thursday (4)
                    itemDate = addSchoolDays(pretestDate, 1, isPM);
                    while ((itemDate.getDay() !== 2 && itemDate.getDay() !== 4) || !isSchoolDay(itemDate, isPM)) {
                        itemDate.setDate(itemDate.getDate() + 1);
                    }
                    
                    // Calculate exact school days elapsed until exam
                    let elapsed = 0;
                    let tempD = new Date(startDate);
                    while (tempD < itemDate) {
                        tempD.setDate(tempD.getDate() + 1);
                        if (isSchoolDay(tempD, isPM)) elapsed++;
                    }
                    dayDisplay = `Day ${elapsed + 1}`;
                } else {
                    itemDate = addSchoolDays(startDate, item.dayOffset, isPM);
                    dayDisplay = `Day ${item.dayOffset + 1}`;
                }

                currentSchedule.push({
                    ...item,
                    date: itemDate,
                    dayDisplay: dayDisplay
                });
            });

            // The completion date is the same as the final exam date
            const examItem = currentSchedule.find(i => i.type === 'exam');
            const completionDate = new Date(examItem.date);
            const totalActualDays = parseInt(examItem.dayDisplay.replace('Day ', ''), 10);

            // 3. Render Output
            renderSchedule(currentSchedule, totalActualDays, completionDate);
        });

        function renderSchedule(scheduleData, totalDays, completionDate) {
            // Hide empty state, show container
            emptyState.classList.add('hidden');
            scheduleContainer.classList.remove('hidden');
            completionEst.classList.remove('hidden');

            // Build Timeline HTML

            scheduleData.forEach((item, index) => {
                const isLast = index === scheduleData.length - 1;
                
                // Determine styling based on type
                let bgColor = 'bg-white';
                let iconColor = 'text-gray-500';
                let badgeColor = 'bg-gray-100 text-gray-800';

                if (item.type === 'start') {
                    bgColor = 'bg-indigo-100';
                    iconColor = 'text-indigo-600';
                    badgeColor = 'bg-indigo-100 text-indigo-800';
                } else if (item.type === 'end') {
                    bgColor = 'bg-green-100';
                    iconColor = 'text-green-600';
                    badgeColor = 'bg-green-100 text-green-800';
                } else if (item.type === 'assignment') {
                    bgColor = 'bg-amber-100';
                    iconColor = 'text-amber-600';
                    badgeColor = 'bg-amber-100 text-amber-800 ring-amber-600/20 ring-1 ring-inset';
                } else if (item.type === 'pretest') {
                    bgColor = 'bg-purple-100';
                    iconColor = 'text-purple-600';
                    badgeColor = 'bg-purple-100 text-purple-800 ring-purple-600/20 ring-1 ring-inset';
                } else if (item.type === 'exam') {
                    bgColor = 'bg-rose-100';
                    iconColor = 'text-rose-600';
                    badgeColor = 'bg-rose-100 text-rose-800 ring-rose-600/20 ring-1 ring-inset';
                } else {
                    bgColor = 'bg-blue-50';
                    iconColor = 'text-blue-500';
                    badgeColor = 'bg-blue-50 text-blue-700';
                }

                html += `
                    <div class="relative pl-8 sm:pl-10 group">
                        <!-- Timeline Dot -->
                        <div class="absolute -left-4 top-1 w-8 h-8 rounded-full border-4 border-white ${bgColor} flex items-center justify-center shadow-sm">
                            <i data-lucide="${item.icon}" class="w-4 h-4 ${iconColor}"></i>
                        </div>
                        
                        <!-- Content Card -->
                        <div class="bg-white border border-gray-100 rounded-xl p-4 sm:p-5 shadow-sm hover:shadow-md transition-shadow">
                            <div class="flex flex-col sm:flex-row sm:justify-between sm:items-start gap-2 mb-2">
                                <h3 class="text-base sm:text-lg font-bold text-gray-900">${item.title}</h3>
                                <div class="flex flex-col sm:items-end text-sm">
                                    <span class="font-medium text-gray-900">${formatDate(item.date)}</span>
                                    <span class="inline-flex items-center rounded-md px-2 py-1 text-xs font-medium mt-1 ${badgeColor} w-fit">
                                        ${item.dayDisplay}
                                    </span>
                                </div>
                            </div>
                            <div class="text-gray-600 text-sm leading-relaxed">${item.desc}</div>
